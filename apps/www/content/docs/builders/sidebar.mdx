## Usage

```tsx
import {
  Sidebar,
  SidebarContent,
  SidebarFooter,
  SidebarHeader,
  SidebarNavItem,
  SidebarSection,
} from "@/components/targetblank/components/sidebar";
import { LayoutDashboard, Settings, Users } from "lucide-react";

export function AppSidebar() {
  return (
    <Sidebar>
      <SidebarHeader>
        <span className="font-semibold text-sm">Acme Inc.</span>
      </SidebarHeader>
      <SidebarContent>
        <SidebarSection label="Main">
          <SidebarNavItem icon={<LayoutDashboard size={16} />} active>
            Dashboard
          </SidebarNavItem>
          <SidebarNavItem icon={<Users size={16} />} badge="12">
            Users
          </SidebarNavItem>
          <SidebarNavItem icon={<Settings size={16} />}>
            Settings
          </SidebarNavItem>
        </SidebarSection>
      </SidebarContent>
      <SidebarFooter>
        <SidebarNavItem>Account</SidebarNavItem>
      </SidebarFooter>
    </Sidebar>
  );
}
```
