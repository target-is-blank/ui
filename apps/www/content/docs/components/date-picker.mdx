## Usage

```tsx
import { DatePicker, DatePickerInput } from "@/components/targetblank/components/date-picker";

// Standalone calendar
<DatePicker value={date} onChange={setDate} />

// Input trigger with popover
<DatePickerInput value={date} onChange={setDate} placeholder="Pick a date" />
```
